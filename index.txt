    const programs = [

        { id: 2, name: "옥스퍼드 아카데미", region: "oxford", type: "degree", price: 4500000, tags: ["명문", "학위"], desc: "PDF 브로셔 분석 결과: 옥스퍼드 대학교 진학 패스웨이 제공, 1:1 튜터링 포함." },
        { id: 3, name: "뉴욕 글로벌 센터", region: "usa", type: "general", price: 3200000, tags: ["미국", "집중학습"], desc: "PDF 브로셔 분석 결과: 맨해튼 위치, 테크니컬 영어 강화, 글로벌 네트워킹 강점." },
        { id: 4, name: "영국 주니어 캠프", region: "london", type: "camp", price: 1800000, tags: ["단기", "청소년"], desc: "PDF 브로셔 분석 결과: 여름 시즌 한정, 기숙사 완비, 현지 체험 중심 커리큘럼." }
    ];

    function renderPrograms(data) {
        const list = document.getElementById('programList');
        list.innerHTML = data.map(p => `
            <div class="program-card" onclick="openModal(${p.id})">
                <div class="card-img">${p.name}</div>
                <div class="card-content">
                    <span class="tag">${p.tags[0]}</span>
                    <h4>${p.name}</h4>
                    <p style="font-size: 0.9rem; color: #64748b;">${p.region.toUpperCase()} | ${p.type}</p>
                    <div class="price">₩ ${p.price.toLocaleString()}</div>
                </div>
            </div>
        `).join('');
    }

    function filterPrograms() {
        const region = document.getElementById('regionFilter').value;
        const type = document.getElementById('typeFilter').value;
        const budget = document.getElementById('budgetInput').value;

        const filtered = programs.filter(p => {
            const matchRegion = region === 'all' || p.region === region;
            const matchType = type === 'all' || p.type === type;
            const matchBudget = !budget || p.price <= parseInt(budget);
            return matchRegion && matchType && matchBudget;
        });
        renderPrograms(filtered);
    }

    function simulateUpload() {
        alert("AI가 PDF 브로셔의 텍스트와 이미지를 분석 중입니다...\n지역: 런던, 학비: 2,500,000원 정보를 추출했습니다.");
        filterPrograms();
    }

    function openModal(id) {
        const p = programs.find(item => item.id === id);
        const modal = document.getElementById('programModal');
        const body = document.getElementById('modalBody');
        body.innerHTML = `
            <h2>${p.name} 상세 정보</h2>
            <p style="color: var(--primary); margin: 1rem 0;">${p.region.toUpperCase()} / ${p.type.toUpperCase()}</p>
            <hr style="margin: 1.5rem 0; border: 0; border-top: 1px solid #e2e8f0;">
            <h3>브로셔 분석 내용</h3>
            <p style="margin-top: 1rem; background: #f1f5f9; padding: 1.5rem; border-radius: 10px;">${p.desc}</p>
            <div class="price" style="font-size: 1.5rem; margin-top: 2rem;">총 비용: ₩ ${p.price.toLocaleString()}</div>
            <button class="cta-button" style="background: var(--primary); color: white; padding: 1rem 2rem; border: none; border-radius: 10px; margin-top: 2rem; cursor: pointer; width: 100%; font-weight: 700;">상담 예약하기</button>
        `;
        modal.style.display = 'flex';
    }

    function closeModal() {
        document.getElementById('programModal').style.display = 'none';
    }

    // 초기 렌더링
    renderPrograms(programs);
